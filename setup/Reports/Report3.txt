--Fazer pesquisa pelo nome ou id?
drop function constructor_list_drivers(constructor_name varchar, forename text, surname text, victories bigint);
create function constructor_list_drivers(constructor_name varchar, out forename text, out surname text, out victories bigint) returns setof record
language plpgsql
as

$$
begin
    return query (
		select d.forename,d.surname, count(d.driverid) as victories
			from driver as d
			join (
					select * from results where constructorid in(select constructorid from constructors where constructorref = constructor_name) and positiontext = '1'
				 ) result on result.driverid = d.driverid
			group by d.driverid
			order by d.driverid
		);
end;
$$;

-- Consultar pilotos do escudeiro
drop function constructor_query_driver(constructor_name varchar, forename text, surname text, dob date, nationality text);
create function constructor_query_driver(constructor_name varchar, out forename text, out surname text, out dob date, out nationality text) returns setof record
language plpgsql
as

$$
begin
    return query (
		select d.forename,d.surname,d.dob,d.nationality
			from driver as d
			where driverid 
			in(	select driverid from results where constructorid in(select constructorid from constructors where constructorref = constructor_name))
			order by driverid
		);
end;
$$;
